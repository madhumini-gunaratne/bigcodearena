export function canMake24(cards: number[]): boolean {
  // Ensure input is valid
  if (cards.length !== 4) return false;

  // Generate all permutations of the four numbers
  const permutations = generatePermutations(cards);

  // For each permutation, try all possible binary operation combinations
  for (const perm of permutations) {
    if (tryAllCombinations(perm)) {
      return true;
    }
  }

  return false;
}

function generatePermutations(arr: number[]): number[][] {
  if (arr.length === 1) return [arr];
  const result: number[][] = [];
  for (let i = 0; i < arr.length; i++) {
    const current = arr[i];
    const remaining = [...arr.slice(0, i), ...arr.slice(i + 1)];
    for (const perm of generatePermutations(remaining)) {
      result.push([current, ...perm]);
    }
  }
  return result;
}

function tryAllCombinations(numbers: number[]): boolean {
  // We will use recursive reduction: combine two numbers at a time
  // Try all possible pairs and operations
  const ops = ['+', '-', '*', '/'];
  const results: number[][] = [];

  // Helper function to evaluate expression string
  const evaluateExpression = (expr: string): number => {
    try {
      return eval(expr);
    } catch (e) {
      return NaN;
    }
  };

  // Generate all possible expressions via recursive combination
  const generateExpressions = (nums: number[], depth: number): string[] => {
    if (depth === 1) {
      return nums.map(num => `${num}`);
    }

    const expressions: string[] = [];
    for (let i = 0; i < nums.length; i++) {
      for (let j = 0; j < nums.length; j++) {
        if (i !== j) {
          for (const op of ops) {
            const newNums = [...nums];
            newNums.splice(i, 1);
            newNums.splice(j, 1);
            const combined = combine(nums[i], nums[j], op);
            const expr = `${nums[i]} ${op} ${nums[j]}`;
            expressions.push(expr);
            // Recursively combine
            const nextExprs = generateExpressions([...newNums, combined], depth - 1);
            expressions.push(...nextExprs);
          }
        }
      }
    }
    return expressions;
  };

  // Try to reduce to one number using all operations
  const allExpressions = generateExpressions(numbers, 4);
  for (const expr of allExpressions) {
    const result = evaluateExpression(expr);
    if (Math.abs(result - 24) < 1e-6) {
      return true;
    }
  }

  return false;
}

function combine(a: number, b: number, op: string): number {

  const [focusedIndex, setFocusedIndex] = useState<number | null>(null);

  const handleFocus = (index: number) => {
    setFocusedIndex(index);
  };

  const handleBlur = () => {
    setFocusedIndex(null);
  };

  return (
    <div className="grid grid-cols-2 gap-4 mt-6">
      {input.map((val, index) => (
        <div
          key={index}
          className={`relative p-4 border-2 rounded-lg text-center transition-all ${
            focusedIndex === index
              ? 'border-indigo-500 bg-indigo-50'
              : 'border-gray-300 hover:border-gray-400'
          }`}
          onFocus={() => handleFocus(index)}
          onBlur={handleBlur}
        >
          <input
            type="number"
            min="1"
            max="9"
            value={val}
            onChange={(e) => onChange(index, e.target.value)}
            className="w-full text-2xl font-bold bg-transparent outline-none"
          />
        </div>
      ))}
    </div>
  );
}
