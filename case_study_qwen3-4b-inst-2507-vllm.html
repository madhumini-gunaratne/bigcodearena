```html
      if (states.length === 0) {

    // Generate Mermaid code from input
    function generateMermaidCode() {
      const states = statesInput.value.trim().split(',').map(s => s.trim());
      const transitions = transitionsInput.value.trim().split('\n').filter(line => line.trim() !== '');
      const startState = startStateSelect.value;
      const endStates = endStatesInput.value.trim().split(',').map(s => s.trim());

      // Validate inputs
      if (!states.length) {
        alert('Please enter at least one state.');
        return null;
      }
      if (!startState) {
        alert('Please select a start state.');
        return null;
      }

      // Validate that start state exists
      if (!states.includes(startState)) {
        alert(`Start state '${startState}' is not in the list of states.`);
        return null;
      }

      // Build Mermaid stateDiagram-v2 syntax
      let mermaidCode = 'stateDiagram-v2\n';
      
      // Add initial state
      mermaidCode += `[*] --> ${startState}\n`;

      // Add transitions
      transitions.forEach(trans => {
        const parts = trans.trim().split(' -> ');
        if (parts.length !== 2) {
          console.warn('Invalid transition format:', trans);
          return;
        }
        const [from, to] = parts;
        const inputMatch = from.match(/(.+) on (.+)/);
        const input = inputMatch ? inputMatch[2] : '';
        
        // If input is present, include it in the transition
        if (input) {
          mermaidCode += `${from} --> ${to} : ${input}\n`;
        } else {
          mermaidCode += `${from} --> ${to}\n`;
        }
      });

      // Add end states (optional)
      if (endStates.length > 0) {

          if (states.includes(endState)) {
            mermaidCode += `${endState} --> [*]\n`;

            console.warn(`End state '${endState}' not found in states.`);
          }
        });
```

```html
          { index: 1, x: x + 50, y: y - 20 }
        ];
      }

      gates.push(gate);
      gatePositions.push({ x, y });

      // Add wire endpoints
      if (type !== 'NOT') {
        gate.outputs = [
          { index: 0, x: x + 50, y: y - 20 },
          { index: 1, x: x + 50, y: y + 20 }
        ];
      } else {
        gate.outputs = [
          { index: 0, x: x + 50, y: y - 20 }
        ];
      }

      // Update the canvas
      render();

      return gate;
    }

    // Function to draw a gate
    function drawGate(gate) {
      const { x, y, type } = gate;
      const width = 100;
      const height = 60;

      ctx.save();
      ctx.translate(x, y);

      // Background
      ctx.fillStyle = '#34495e';
      ctx.fillRect(-width/2, -height/2, width, height);

      // Border
      ctx.strokeStyle = '#2c3e50';
      ctx.lineWidth = 2;
      ctx.strokeRect(-width/2, -height/2, width, height);

      // Draw gate body
      ctx.fillStyle = '#2c3e50';
      ctx.beginPath();
      ctx.moveTo(-width/2, -height/2);
      ctx.lineTo(width/2, -height/2);
      ctx.lineTo(width/2, height/2);
      ctx.lineTo(-width/2, height/2);
      ctx.closePath();
      ctx.fill();

      // Draw inputs and outputs
      const inputY = -height/2 + 10;
      const outputY = height/2 - 10;

      if (type === 'NOT') {
        // NOT gate: one input, one output
        ctx.fillStyle = '#e74c3c';
        ctx.fillRect(-width/2 + 10, inputY, 20, 10);
        ctx.fillStyle = '#2ecc71';
        ctx.fillRect(-width/2 + 10, outputY, 20, 10);
      } else {
        // Two-input gates
        ctx.fillStyle = '#e74c3c';
        ctx.fillRect(-width/2 + 10, inputY, 20, 10);
        ctx.fillRect(width/2 - 30, inputY, 20, 10);
        ctx.fillStyle = '#2ecc71';
        ctx.fillRect(width/2 - 30, outputY, 20, 10);
      }

      // Draw gate label
      ctx.fillStyle = 'white';
      ctx.font = 'bold 14px Arial';
      ctx.textAlign = 'center';
      ctx.fillText(gate.type, 0, -height/2 + 30);

      ctx.restore();
    }

    // Function to draw a wire
    function drawWire(start, end) {
      const { x: startX, y: startY } = start;
      const { x: endX, y: endY } = end;

      // Simple line
      ctx.beginPath();
      ctx.moveTo(startX, startY);
      ctx.lineTo(endX, endY);
      ctx.strokeStyle = '#3498db';
      ctx.lineWidth = 2;
      ctx.stroke();

      // Add a small dot at the end
      ctx.beginPath();
      ctx.arc(endX, endY, 4, 0, Math.PI * 2);
      ctx.fillStyle = '#3498db';
      ctx.fill();
    }

    // Function to render the entire circuit
    function render() {
      ctx.clearRect(0, 0, canvas.width, canvas.height);

      // Draw all gates
      gates.forEach(gate => {
        drawGate(gate);
      });

      // Draw all wires
      wires.forEach(wire => {
        drawWire(wire.start, wire.end);
      });

      // Update output
      updateOutput();
    }

    // Update the output based on current inputs and gate logic
    function updateOutput() {
      let finalOutput = 0;

      // We will simulate the circuit from inputs to outputs
      // For simplicity, we assume only one output at the end
      // In a real circuit, we would traverse the gates in order

      // We'll simulate by evaluating each gate in order (by position)
      // We need to determine the input values for each gate
      const inputMap = {
        input1: inputValues.input1.checked,
        input2: inputValues.input2.checked,
        input3: inputValues.input3.checked,
        input4: inputValues.input4.checked
      };

      // We'll create a map of input values to gate inputs
      const gateInputValues = {};

      // Evaluate each gate
      gates.forEach(gate => {
        let value = null;
        const inputs = gate.inputs;

        if (gate.type === 'NOT') {
          // Get input value from the input map (we'll map by index)
          const inputIndex = 0;
          const inputName = `input${inputIndex + 1}`;
          if (inputMap[inputName]) {
            value = inputMap[inputName];
          } else {
            value = 0;
          }
          gate.outputValue = gate.fn(value);
        } else if (gate.type === 'AND' || gate.type === 'OR' || gate.type === 'NAND' || gate.type === 'NOR' || gate.type === 'XOR' || gate.type === 'XNOR') {
          const a = inputs[0].index === 0 ? inputMap.input1 : inputMap.input2;
          const b = inputs[1].index === 0 ? inputMap.input1 : inputMap.input2;
          value = gate.fn(a, b);
          gate.outputValue = value;
        }

        // Store output value for this gate
        gateInputValues[gate.id] = value;
      });

      // For now, we just show the last gate's output
      // In a real circuit, we'd have to traverse the connections
      // But for simplicity, we just use the last gate
      finalOutput = gates[gates.length - 1]?.outputValue || 0;

      document.getElementById('output-value').textContent = finalOutput;
    }

    // Handle drag and drop
    document.addEventListener('mousedown', (e) => {
      const rect = canvas.getBoundingClientRect();
      mouseX = e.clientX - rect.left;
      mouseY = e.clientY - rect.top;

      // Check if clicking on a gate
      gates.forEach(gate => {
        const { x, y } = gate;
        const width = 100;
        const height = 60;
        if (mouseX >= x - width/2 && mouseX <= x + width/2 &&
            mouseY >= y - height/2 && mouseY <= y + height/2) {
          selectedGate = gate;
          isDragging = true;
          return;
        }
      });

      // Check if clicking on a wire (simplified)
      // For now, we just allow creating a gate on click
      if (!selectedGate) {
        const x = mouseX;
        const y = mouseY;
        const type = 'AND'; // default
        createGate(type, x, y);
      }
    });

    document.addEventListener('mousemove', (e) => {
      if (isDragging && selectedGate) {
        selectedGate.x = mouseX;
        selectedGate.y = mouseY;
        render();
      }
    });

    document.addEventListener('mouseup', () => {
      isDragging = false;
      selectedGate = null;
    });

    // Handle gate button clicks
    document.querySelectorAll('.gate-btn').forEach(btn => {
      btn.addEventListener('click', () => {
        const type = btn.getAttribute('data-gate');
        const x = mouseX;
        const y = mouseY;
        createGate(type, x, y);
      });
    });

    // Update simulation when input checkboxes change
    inputValues.input1.addEventListener('change', updateOutput);
    inputValues.input2.addEventListener('change', updateOutput);
    inputValues.input3.addEventListener('change', updateOutput);
    inputValues.input4.addEventListener('change', updateOutput);

    // Handle window resize
    window.addEventListener('resize', () => {
      canvas.width = window.innerWidth * 0.95;
      canvas.height = 600;
      render();
    });

    // Initial render
    render();
  </script>
</body>
</html>
```

### How to Use This Simulator

1. **Open the HTML file in a browser** (no server needed).
2. **Click on a gate button** (e.g., AND, OR) to place a gate on the canvas.
3. **Drag the gate** to reposition it.
4. **Toggle the input checkboxes** to change input values.
5. **Observe the output** at the bottom.

---

### Limitations (Educational Notes)

- This is a **simplified simulation**. It doesn't fully handle complex circuit topologies (like multiple gate connections).
- It only evaluates the last gate's output. In a real circuit, you'd need to traverse the network.
- Wires are not fully connected (we don't have a full wire connection system).
- No visual feedback for signal flow (like active wires or delays).
